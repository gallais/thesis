\declaretheorem[
  name=Convention,
]{thmconvene}

\ExplSyntaxOn
\NewDocumentEnvironment{convention}{o}
 {
  \int_gincr:N \g_thesis_convention_int
  \IfNoValueTF{#1}
   {
    \thesis_restatable:x { \int_to_Alph:n { \g_thesis_convention_int } }
   }
   {
    \thesis_restatable:nx { #1 } { \int_to_Alph:n { \g_thesis_convention_int } }
   }
 }
 {
  \endrestatable
  \seq_gput_right:Nx \g_thesis_convention_seq
   {
    convene \int_to_Alph:n { \g_thesis_convention_int }
   }
 }

\NewDocumentCommand{\allconventions}{}
 {
  \seq_map_inline:Nn \g_thesis_convention_seq
   {
    \use:c { ##1 } *
   }
 }

\int_new:N \g_thesis_convention_int
\seq_new:N \g_thesis_convention_seq

\cs_new_protected:Npn \thesis_restatable:n #1
 {
  \restatable{thmconvene}{convene#1}
 }
\cs_generate_variant:Nn \thesis_restatable:n { x }
\cs_new_protected:Npn \thesis_restatable:nn #1 #2
 {
  \restatable[#1]{thmconvene}{convene#2}
 }
\cs_generate_variant:Nn \thesis_restatable:nn { nx }
\ExplSyntaxOff
